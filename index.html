Welcome dmuka.Popup Documentation
Default Popup 
function showDefaulPopup() {
    var popup = new dmuka.Popup({
        width: "250px",
        height: "100px",
        header: {
            text: "My Popup"
        },
        content: {
            htmlOrChild: "Hello Word"
        }
    });
    popup.open();
}

Disable Footer or Header 
function disablePopup(headerEnable, footerEnable) {
    var popup = new dmuka.Popup({
        width: "250px",
        height: "100px",
        header: {
            text: "Question",
            enable: headerEnable
        },
        content: {
            htmlOrChild: "Do you want continue?"
        },
        footer: {
            enable: footerEnable,
            buttons: [
                {
                    text: "Yes",
                    id: "yes"
                },
                {
                    text: "No",
                    id: "no"
                },
                {
                    text: "Cancel",
                    id: "cancel",
                    focus: true
                }
            ]
        }
    });
    popup.open();
}
Custom Footer 
function oneButtonPopup() {
    var popup = new dmuka.Popup({
        width: "250px",
        height: "100px",
        header: {
            text: "Message"
        },
        content: {
            htmlOrChild: "Process complated..."
        },
        footer: {
            buttons: [
                {
                    text: "Ok",
                    id: "ok",
                    focus: true
                }
            ]
        },
        onSubmitFooter: function (footerButtonId) {
            alert("Clicked ID = " + footerButtonId);
        }
    });
    popup.open();
}
